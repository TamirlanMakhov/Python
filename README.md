```python
    row = [int(x) for x in input().split()]
    matrix.append(row)
    
for i in range(n):
    for j in range(n):
        if i < j:
            if i < n - 1 - j:
                up.append(matrix[i][j])
            elif i > n - 1 - j:
                right.append(matrix[i][j])
```

```python
from functools import reduce as r

eval = lambda coef, x: r(lambda v, c: c + v * x, map(int, coef))
```
